import bpy
import tensorflow as tf
import numpy as np

# Extract mesh data from Blender
def extract_mesh_data(obj):
    vertices = np.array([v.co for v in obj.data.vertices])
    faces = np.array([p.vertices for p in obj.data.polygons])
    return {
        'vertices': vertices,
        'faces': faces,
        'name': obj.name
    }

# Process for machine learning
def prepare_for_ml(mesh_data):
    # Normalize vertices
    vertices = mesh_data['vertices']
    vertices_normalized = (vertices - vertices.min()) / (vertices.max() - vertices.min())
    
    # Create input tensors
    input_tensor = tf.convert_to_tensor(vertices_normalized, dtype=tf.float32)
    return input_tensor

# Train model on mesh patterns
def train_model(input_tensors, labels):
    model = tf.keras.Sequential([
        tf.keras.layers.Dense(128, activation='relu'),
        tf.keras.layers.Dense(64, activation='relu'),
        tf.keras.layers.Dense(32, activation='relu'),
        tf.keras.layers.Dense(3, activation='linear')
    ])
    
    model.compile(
        optimizer='adam',
        loss='mse'
    )
    
    model.fit(input_tensors, labels, epochs=100)
    return model
